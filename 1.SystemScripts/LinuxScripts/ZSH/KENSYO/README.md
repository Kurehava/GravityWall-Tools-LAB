# 替换老主题文件方法
```
# User
curl https://raw.githubusercontent.com/Kurehava/GravityWall-Tools-LAB/refs/heads/main/1.SystemScripts/LinuxScripts/ZSH/KENSYO/kensyo_theme.zsh-theme > /home/$(whoami)/.oh-my-zsh/themes/kensyo.zsh-theme && source /home/$(whoami)/.zshrc

# Root
curl https://raw.githubusercontent.com/Kurehava/GravityWall-Tools-LAB/refs/heads/main/1.SystemScripts/LinuxScripts/ZSH/KENSYO/kensyo_theme.zsh-theme > /root/.oh-my-zsh/themes/kensyo.zsh-theme && /root/.zshrc
```
